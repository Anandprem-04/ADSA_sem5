    public int findMin(int[] nums) {
        if(nums.length == 1) return nums[0];
        int l = 0 , h = nums.length-1;
        int ans = Integer.MAX_VALUE;
        while(l<=h){
            int mid = l + (h-l)/2;
            if(nums[mid]>=nums[l]){
                ans = Math.min(ans, nums[l]);
                l = mid+1;
            }else{
                ans = Math.min(ans, nums[mid]);
                h = mid-1;
            }
        }
        return ans;
    }
