    public Node copyRandomList(Node head) {
        if(head == null) return null;

        Node temp = head;
        while(temp != null){
            Node newNode = new Node(temp.val);
            newNode.next = temp.next;
            temp.next = newNode;
            temp = temp.next.next;
        }
        temp = head;
        while(temp != null){
            if(temp.random != null){
                temp.next.random = temp.random.next;
            }
            temp = temp.next.next;
        }
        temp = head;
        Node dummy = head.next;
        while(temp != null){
            Node copy = temp.next;
            temp.next = copy.next;
            if(copy.next != null){
                copy.next = copy.next.next;
            }
            temp = temp.next;
        }
        return dummy;
    }
